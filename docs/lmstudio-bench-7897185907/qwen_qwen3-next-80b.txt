    <!-- Script for Kanban Board Functionality -->
    <script>
        // Namespace for our application to avoid global variable conflicts
        const KANBAN = {
            // Storage key for localStorage persistence
            STORAGE_KEY: 'kanban-board-state',
            
            // Column definitions with their IDs and color classes
            COLUMNS: {
                backlog: { id: 'backlog', name: 'Backlog', colorClass: 'bg-gray-50' },
                inProgress: { id: 'in-progress', name: 'In Progress', colorClass: 'bg-blue-50' },
                review: { id: 'review', name: 'Review', colorClass: 'bg-yellow-50' },
                done: { id: 'done', name: 'Done', colorClass: 'bg-green-50' }
            },
            
            // Initialize the board
            init() {
                // Load saved state from localStorage or create default data if none exists
                this.loadState();
                
                // Set up event listeners for adding cards, drag and drop functionality
                this.setupEventListeners();
                
                // Render the board with loaded data or dummy data if empty
                this.renderBoard();
            },
            
            // Load board state from localStorage or create default data
            loadState() {
                try {
                    const savedState = localStorage.getItem(this.STORAGE_KEY);
                    
                    if (savedState) {
                        this.state = JSON.parse(savedState);
                        
                        // Ensure all columns exist
                        for (const columnId in this.COLUMNS) {
                            if (!this.state[columnId]) {
                                this.state[columnId] = [];
                            }
                        }
                    } else {
                        // Create default dummy data if no state exists
                        this.state = {
                            backlog: [
                                { id: 'card-1', title: 'Design the new dashboard layout', description: 'Create wireframes for the admin panel', createdAt: new Date().toISOString() },
                                { id: 'card-2', title: 'Implement user authentication system', description: 'Set up login, registration, and password reset functionality', createdAt: new Date().toISOString() },
                                { id: 'card-3', title: 'Optimize database queries', description: 'Reduce load time by optimizing SQL queries', createdAt: new Date().toISOString() }
                            ],
                            'in-progress': [
                                { id: 'card-4', title: 'Develop API endpoints for frontend', description: 'Create RESTful APIs to handle data requests', createdAt: new Date().toISOString() },
                                { id: 'card-5', title: 'Write unit tests for core components', description: 'Ensure 90% test coverage for critical functionality', createdAt: new Date().toISOString() }
                            ],
                            review: [
                                { id: 'card-6', title: 'Review UI/UX design mockups', description: 'Gather feedback from stakeholders on the new design', createdAt: new Date().toISOString() }
                            ],
                            done: [
                                { id: 'card-7', title: 'Set up project repository', description: 'Initialize git repo and configure CI/CD pipeline', createdAt: new Date().toISOString() },
                                { id: 'card-8', title: 'Configure development environment', description: 'Install required dependencies and set up local server', createdAt: new Date().toISOString() }
                            ]
                        };
                    }
                } catch (error) {
                    console.error('Error loading board state:', error);
                    // If there's an error, create fresh default state
                    this.state = {
                        backlog: [
                            { id: 'card-1', title: 'Design the new dashboard layout', description: 'Create wireframes for the admin panel', createdAt: new Date().toISOString() },
                            { id: 'card-2', title: 'Implement user authentication system', description: 'Set up login, registration, and password reset functionality', createdAt: new Date().toISOString() },
                            { id: 'card-3', title: 'Optimize database queries', description: 'Reduce load time by optimizing SQL queries', createdAt: new Date().toISOString() }
                        ],
                        'in-progress': [
                            { id: 'card-4', title: 'Develop API endpoints for frontend', description: 'Create RESTful APIs to handle data requests', createdAt: new Date().toISOString() },
                            { id: 'card-5', title: 'Write unit tests for core components', description: 'Ensure 90% test coverage for critical functionality', createdAt: new Date().toISOString() }
                        ],
                        review: [
                            { id: 'card-6', title: 'Review UI/UX design mockups', description: 'Gather feedback from stakeholders on the new design', createdAt: new Date().toISOString() }
                        ],
                        done: [
                            { id: 'card-7', title: 'Set up project repository', description: 'Initialize git repo and configure CI/CD pipeline', createdAt: new Date().toISOString() },
                            { id: 'card-8', title: 'Configure development environment', description: 'Install required dependencies and set up local server', createdAt: new Date().toISOString() }
                        ]
                    };
                }
                
                // Save state after loading (in case we needed to fix it)
                this.saveState();
            },
            
            // Save board state to localStorage
            saveState() {
                try {
                    localStorage.setItem(this.STORAGE_KEY, JSON.stringify(this.state));
                } catch (error) {
                    console.error('Error saving board state:', error);
                }
            },
            
            // Set up all event listeners for the application
            setupEventListeners() {
                // Add card button click handlers for all columns
                document.querySelectorAll('.add-card-btn').forEach(button => {
                    button.addEventListener('click', (e) => {
                        const columnId = e.target.closest('.column').id;
                        this.addCard(columnId);
                    });
                });
                
                // Drag and drop event listeners for the entire board
                document.addEventListener('dragstart', (e) => {
                    if (e.target.classList.contains('card')) {
                        e.dataTransfer.setData('text/plain', e.target.id);
                        // Add dragging class to card for visual feedback
                        e.target.classList.add('dragging');
                    }
                });
                
                document.addEventListener('dragover', (e) => {
                    e.preventDefault(); // Required to allow drop
                    const target = e.target;
                    
                    // If we're over a cards container, show visual feedback
                    if (target.classList.contains('cards-container')) {
                        target.style.backgroundColor = '#e0f7fe';
                    }
                });
                
                document.addEventListener('dragleave', (e) => {
                    const target = e.target;
                    
                    // Remove visual feedback when leaving
                    if (target.classList.contains('cards-container')) {
                        target.style.backgroundColor = '';
                    }
                });
                
                document.addEventListener('drop', (e) => {
                    e.preventDefault();
                    
                    // Get the card ID from data transfer
                    const cardId = e.dataTransfer.getData('text/plain');
                    
                    // If no card ID, return
                    if (!cardId) return;
                    
                    const draggedCard = document.getElementById(cardId);
                    
                    // If card doesn't exist, return
                    if (!draggedCard) return;
                    
                    // Get the target column (the column that received the drop)
                    const targetColumn = e.target.closest('.column');
                    
                    // If no column was targeted, return
                    if (!targetColumn) {
                        e.target.style.backgroundColor = '';
                        return;
                    }
                    
                    // Get the target column ID
                    const targetColumnId = targetColumn.id;
                    
                    // Find source column (where the card came from)
                    const sourceColumnId = draggedCard.closest('.column').id;
                    
                    // If the card is being dropped in the same column, do nothing
                    if (sourceColumnId === targetColumnId) {
                        e.target.style.backgroundColor = '';
                        return;
                    }
                    
                    // Remove the card from its current position
                    draggedCard.remove();
                    
                    // Add it to the target column's cards container
                    const targetCardsContainer = targetColumn.querySelector('.cards-container');
                    
                    // If the drop is not on a cards container, find the nearest one
                    if (!targetCardsContainer) {
                        const potentialTarget = e.target.closest('.cards-container');
                        if (potentialTarget) {
                            targetCardsContainer.appendChild(draggedCard);
                        }
                    } else {
                        // Add to the end of the target column
                        targetCardsContainer.appendChild(draggedCard);
                    }
                    
                    // Update state with new column position
                    this.moveCard(cardId, sourceColumnId, targetColumnId);
                    
                    // Trigger confetti animation if card moved to Done column
                    if (targetColumnId === 'done') {
                        this.triggerConfetti();
                    }
                    
                    // Update column counters
                    this.updateColumnCounters();
                    
                    e.target.style.backgroundColor = '';
                });
                
                // Handle card deletion with confirmation
                document.addEventListener('click', (e) => {
                    if (e.target.classList.contains('delete-card-btn')) {
                        const cardId = e.target.closest('.card').id;
                        
                        // Show custom confirmation dialog instead of alert
                        if (confirm('Are you sure you want to delete this card? This action cannot be undone.')) {
                            this.deleteCard(cardId);
                        }
                    }
                });
                
                // Handle card content editing (in-place)
                document.addEventListener('click', (e) => {
                    if (e.target.classList.contains('card-content') && !e.target.closest('.delete-card-btn')) {
                        // Only allow editing if it's not a delete button click
                        const contentElement = e.target;
                        
                        // Don't edit if it's already being edited
                        if (contentElement.hasAttribute('data-editing')) {
                            return;
                        }
                        
                        // Mark as editing
                        contentElement.setAttribute('data-editing', 'true');
                        contentElement.contentEditable = true;
                        
                        // Focus the element
                        contentElement.focus();
                        
                        // Select all text when editing starts
                        const range = document.createRange();
                        range.selectNodeContents(contentElement);
                        const selection = window.getSelection();
                        selection.removeAllRanges();
                        selection.addRange(range);
                    }
                });
                
                // Handle blur on content editing (save changes)
                document.addEventListener('blur', (e) => {
                    if (e.target.classList.contains('card-content') && e.target.hasAttribute('data-editing')) {
                        const contentElement = e.target;
                        
                        // Remove editing attribute
                        contentElement.removeAttribute('data-editing');
                        contentElement.contentEditable = false;
                        
                        // Update the card data in state
                        const cardId = contentElement.closest('.card').id;
                        const columnId = contentElement.closest('.column').id;
                        
                        // Update the card title in state
                        const card = this.state[columnId].find(card => card.id === cardId);
                        
                        if (card) {
                            // Check if the content changed
                            const newContent = contentElement.textContent.trim();
                            
                            if (newContent !== card.title) {
                                card.title = newContent;
                                this.saveState();
                            }
                        }
                    }
                });
            },
            
            // Add a new card to the specified column
            addCard(columnId) {
                // Generate unique ID for the new card
                const id = `card-${Date.now()}-${Math.random().toString(36).substr(2, 9)}`;
                
                // Create new card data
                const newCard = {
                    id: id,
                    title: 'New Task',
                    description: '',
                    createdAt: new Date().toISOString()
                };
                
                // Add card to the state
                if (!this.state[columnId]) {
                    this.state[columnId] = [];
                }
                
                // Add card to the beginning of the column
                this.state[columnId].unshift(newCard);
                
                // Save state to localStorage
                this.saveState();
                
                // Render the updated board
                this.renderBoard();
            },
            
            // Move a card from one column to another
            moveCard(cardId, sourceColumnId, targetColumnId) {
                // Remove card from source column
                const sourceIndex = this.state[sourceColumnId].findIndex(card => card.id === cardId);
                if (sourceIndex > -1) {
                    this.state[sourceColumnId].splice(sourceIndex, 1);
                }
                
                // Add card to target column
                if (!this.state[targetColumnId]) {
                    this.state[targetColumnId] = [];
                }
                
                // Add to the end of target column
                this.state[targetColumnId].push(
                    this.state[sourceColumnId].find(card => card.id === cardId) || 
                    { id: cardId, title: 'Unknown Card', description: '', createdAt: new Date().toISOString() }
                );
                
                // Save state
                this.saveState();
            },
            
            // Delete a card from the board
            deleteCard(cardId) {
                // Find which column contains this card and remove it
                for (const columnId in this.state) {
                    const index = this.state[columnId].findIndex(card => card.id === cardId);
                    
                    if (index > -1) {
                        this.state[columnId].splice(index, 1);
                        break;
                    }
                }
                
                // Save state
                this.saveState();
                
                // Render the updated board
                this.renderBoard();
            },
            
            // Trigger confetti animation when card is moved to Done
            triggerConfetti() {
                const count = 200;
                const defaults = {
                    origin: { y: 0.7 }
                };
                
                // Create confetti effect
                function fire(particleRatio, opts) {
                    confetti({
                        ...defaults,
                        ...opts,
                        particleCount: Math.floor(count * particleRatio)
                    });
                }
                
                // Create a burst of confetti
                fire(0.25, {
                    spread: 26,
                    startVelocity: 55
                });
                
                fire(0.2, {
                    spread: 60
                });
                
                fire(0.35, {
                    spread: 100,
                    decay: 0.91,
                    scalar: 0.8
                });
                
                fire(0.1, {
                    spread: 120,
                    startVelocity: 25,
                    decay: 0.92,
                    scalar: 1.2
                });
                
                fire(0.1, {
                    spread: 120,
                    startVelocity: 45
                });
            },
            
            // Update the counter badges in each column header
            updateColumnCounters() {
                for (const columnId in this.COLUMNS) {
                    const column = document.getElementById(columnId);
                    if (column) {
                        const cardsContainer = column.querySelector('.cards-container');
                        const countElement = column.querySelector('span.text-xs');
                        
                        if (countElement && cardsContainer) {
                            const count = cardsContainer.children.length;
                            countElement.textContent = count;
                            
                            // Update color of counter based on column
                            if (columnId === 'backlog') {
                                countElement.classList.remove('bg-blue-300', 'bg-yellow-300', 'bg-green-300');
                                countElement.classList.add('bg-gray-300');
                            } else if (columnId === 'in-progress') {
                                countElement.classList.remove('bg-gray-300', 'bg-yellow-300', 'bg-green-300');
                                countElement.classList.add('bg-blue-300');
                            } else if (columnId === 'review') {
                                countElement.classList.remove('bg-gray-300', 'bg-blue-300', 'bg-green-300');
                                countElement.classList.add('bg-yellow-300');
                            } else if (columnId === 'done') {
                                countElement.classList.remove('bg-gray-300', 'bg-blue-300', 'bg-yellow-300');
                                countElement.classList.add('bg-green-300');
                            }
                        }
                    }
                }
            },
            
            // Render the entire board based on current state
            renderBoard() {
                // Clear all existing cards from UI
                document.querySelectorAll('.cards-container').forEach(container => {
                    container.innerHTML = '';
                });
                
                // For each column, render its cards
                for (const columnId in this.state) {
                    const column = document.getElementById(columnId);
                    
                    if (column) {
                        const cardsContainer = column.querySelector('.cards-container');
                        
                        // Clear existing cards
                        cardsContainer.innerHTML = '';
                        
                        // Render each card in this column
                        this.state[columnId].forEach(card => {
                            const cardElement = this.createCardElement(card, columnId);
                            cardsContainer.appendChild(cardElement);
                        });
                    }
                }
                
                // Update counters after rendering
                this.updateColumnCounters();
            },
            
            // Create a card DOM element from card data
            createCardElement(card, columnId) {
                // Create card container div
                const cardDiv = document.createElement('div');
                cardDiv.id = card.id;
                cardDiv.className = 'card bg-white rounded-lg shadow-md p-4 cursor-move transition-all duration-300 hover:shadow-lg transform hover:-translate-y-1';
                
                // Add draggable attribute
                cardDiv.draggable = true;
                
                // Create title element (editable)
                const titleElement = document.createElement('div');
                titleElement.className = 'card-content font-bold text-gray-800 mb-2 text-sm md:text-base';
                titleElement.textContent = card.title || 'Untitled Task';
                
                // Make it editable and add some styling
                titleElement.contentEditable = false;
                titleElement.style.wordWrap = 'break-word';
                
                // Create description element
                const descElement = document.createElement('div');
                descElement.className = 'card-content text-gray-600 mb-3 text-xs md:text-sm';
                descElement.textContent = card.description || '';
                descElement.style.wordWrap = 'break-word';
                
                // Create timestamp element
                const timeElement = document.createElement('div');
                timeElement.className = 'text-xs text-gray-400 mt-2';
                
                // Format date
                const createdAt = new Date(card.createdAt);
                timeElement.textContent = createdAt.toLocaleDateString() + ' at ' + createdAt.toLocaleTimeString([], { hour: '2-digit', minute: '2-digit' });
                
                // Create delete button
                const deleteBtn = document.createElement('button');
                deleteBtn.className = 'delete-card-btn text-red-500 hover:text-red-700 ml-auto p-1 rounded-full transition-colors duration-200 hover:bg-red-50';
                deleteBtn.innerHTML = `
                    <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" viewBox="0 0 20 20" fill="currentColor">
                        <path fill-rule="evenodd" d="M9 2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0 11-2 0V8zm5-1a1 1 0 00-1 1v6a1 1 0 102 0V8a1 1 0 00-1-1z" clip-rule="evenodd"/>
                    </svg>
                `;
                
                // Assemble the card
                cardDiv.appendChild(titleElement);
                cardDiv.appendChild(descElement);
                cardDiv.appendChild(timeElement);
                cardDiv.appendChild(deleteBtn);
                
                return cardDiv;
            }
        };
        
        // Initialize the Kanban board when DOM is loaded
        document.addEventListener('DOMContentLoaded', () => {
            KANBAN.init();
        });
    </script>
