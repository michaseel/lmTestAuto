    <script>
        // ================= STATE MANAGEMENT =================
        const STORAGE_KEY = 'kanban_board_my_namespace_v1';
        let tasks = [];
        let taskToDeleteId = null;

        // Dummy Data for first load
        const dummyData = [
            { id: 't1', title: 'Design System Draft', description: 'Create color palette and typography scale.', column: 'backlog' },
            { id: 't2', title: 'Setup Repo', description: 'Initialize Git and install dependencies.', column: 'done' },
            { id: 't3', title: 'Login Page', description: 'Implement authentication form layout.', column: 'todo' },
            { id: 't4', title: 'API Integration', description: 'Connect frontend to user endpoints.', column: 'review' },
            { id: 't5', title: 'Unit Tests', description: 'Write tests for utility functions.', column: 'todo' },
        ];

        // Initialize App
        function init() {
            const stored = localStorage.getItem(STORAGE_KEY);
            if (stored) {
                try {
                    tasks = JSON.parse(stored);
                } catch (e) {
                    tasks = dummyData;
                }
            } else {
                tasks = [...dummyData];
                saveData();
            }
            renderBoard();
        }

        function saveData() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(tasks));
        }

        function resetBoard() {
            if(confirm("Reset to default demo data?")) {
                tasks = [...dummyData];
                saveData();
                renderBoard();
            }
        }

        // ================= RENDERING =================
        function renderBoard() {
            // Clear all columns
            ['backlog', 'todo', 'review', 'done'].forEach(colId => {
                const container = document.getElementById(`col-${colId}`);
                container.innerHTML = '';
                // Update Badge Count
                const count = tasks.filter(t => t.column === colId).length;
                container.parentElement.querySelector('.count-badge').textContent = count;
            });

            // Render Tasks
            tasks.forEach(task => {
                const card = createCardElement(task);
                const container = document.getElementById(`col-${task.column}`);
                if (container) container.appendChild(card);
            });
        }

        function createCardElement(task) {
            const div = document.createElement('div');
            div.className = 'kanban-card bg-white p-4 rounded-lg shadow-sm border border-slate-200 hover:shadow-md mb-3 relative group';
            div.draggable = true;
            div.dataset.id = task.id;

            // Tag color based on column (visual flair)
            let tagColor = 'bg-slate-100 text-slate-600';
            if(task.column === 'todo') tagColor = 'bg-blue-50 text-blue-600';
            if(task.column === 'review') tagColor = 'bg-purple-50 text-purple-600';
            if(task.column === 'done') tagColor = 'bg-emerald-50 text-emerald-600';

            div.innerHTML = `
                <div class="flex justify-between items-start mb-2">
                    <span class="text-xs font-bold px-2 py-1 rounded ${tagColor} uppercase tracking-wider" style="font-size: 0.65rem;">${task.column}</span>
                    <button class="delete-btn opacity-0 group-hover:opacity-100 text-slate-400 hover:text-red-500 transition-opacity p-1" title="Delete">
                        <svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor">
                            <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" />
                        </svg>
                    </button>
                </div>
                <h4 class="font-semibold text-slate-800 mb-1 leading-tight">${escapeHtml(task.title)}</h4>
                <p class="text-sm text-slate-500 line-clamp-2">${escapeHtml(task.description)}</p>
            `;

            // Event Listeners
            div.addEventListener('dragstart', handleDragStart);
            div.addEventListener('dragend', handleDragEnd);
            
            // Edit on click (excluding delete button)
            div.addEventListener('click', (e) => {
                if (e.target.closest('.delete-btn')) return;
                openTaskModal(task.column, task.id);
            });

            // Delete handler
            const deleteBtn = div.querySelector('.delete-btn');
            deleteBtn.addEventListener('click', (e) => {
                e.stopPropagation();
                openDeleteModal(task.id);
            });

            return div;
        }

        // Helper to prevent XSS
        function escapeHtml(text) {
            if (!text) return '';
            return text
                .replace(/&/g, "&amp;")
                .replace(/</g, "&lt;")
                .replace(/>/g, "&gt;")
                .replace(/"/g, "&quot;")
                .replace(/'/g, "&#039;");
        }

        // ================= DRAG AND DROP LOGIC =================
        let draggedItem = null;

        function handleDragStart(e) {
            draggedItem = this;
            e.dataTransfer.setData('text/plain', this.dataset.id);
            e.dataTransfer.effectAllowed = 'move';
            setTimeout(() => this.classList.add('dragging'), 0);
        }

        function handleDragEnd(e) {
            this.classList.remove('dragging');
            draggedItem = null;
            // Remove drag-over styles from all columns
            document.querySelectorAll('.task-container').forEach(col => {
                col.classList.remove('drag-over');
            });
        }

        // Set up drop zones
        document.querySelectorAll('.column-container').forEach(column => {
            const container = column.querySelector('.task-container');

            column.addEventListener('dragover', (e) => {
                e.preventDefault(); // Necessary to allow dropping
                container.classList.add('drag-over');
            });

            column.addEventListener('dragleave', (e) => {
                // Only remove if leaving the column completely, not entering a child
                if (!column.contains(e.relatedTarget)) {
                    container.classList.remove('drag-over');
                }
            });

            column.addEventListener('drop', (e) => {
                e.preventDefault();
                container.classList.remove('drag-over');
                
                const id = e.dataTransfer.getData('text/plain');
                const targetColumn = column.dataset.column;
                
                if (id && targetColumn) {
                    moveTask(id, targetColumn);
                }
            });
        });

        function moveTask(id, newColumn) {
            const taskIndex = tasks.findIndex(t => t.id === id);
            if (taskIndex > -1 && tasks[taskIndex].column !== newColumn) {
                // If moving to 'done', trigger confetti
                if (newColumn === 'done') {
                    triggerConfetti();
                }

                tasks[taskIndex].column = newColumn;
                saveData();
                renderBoard();
            }
        }

        function triggerConfetti() {
            const duration = 1000;
            const end = Date.now() + duration;

            (function frame() {
                confetti({
                    particleCount: 5,
                    angle: 60,
                    spread: 55,
                    origin: { x: 0 },
                    colors: ['#34D399', '#60A5FA', '#F472B6']
                });
                confetti({
                    particleCount: 5,
                    angle: 120,
                    spread: 55,
                    origin: { x: 1 },
                    colors: ['#34D399', '#60A5FA', '#F472B6']
                });

                if (Date.now() < end) {
                    requestAnimationFrame(frame);
                }
            }());
        }

        // ================= MODAL & CRUD LOGIC =================
        
        // Open Modal for Create or Edit
        function openTaskModal(columnId, taskId = null) {
            const modal = document.getElementById('taskModal');
            const titleInput = document.getElementById('taskInputTitle');
            const descInput = document.getElementById('taskInputDesc');
            const idInput = document.getElementById('taskId');
            const colInput = document.getElementById('taskColumn');
            const modalTitle = document.getElementById('modalTitle');

            // Reset form
            document.getElementById('taskForm').reset();
            colInput.value = columnId;

            if (taskId) {
                // Edit Mode
                const task = tasks.find(t => t.id === taskId);
                if (task) {
                    modalTitle.textContent = 'Edit Card';
                    idInput.value = task.id;
                    titleInput.value = task.title;
                    descInput.value = task.description;
                }
            } else {
                // Create Mode
                modalTitle.textContent = 'Add New Card';
                idInput.value = '';
            }

            modal.classList.remove('hidden');
            modal.classList.add('flex');
            titleInput.focus();
        }

        function closeModal(modalId) {
            const modal = document.getElementById(modalId);
            modal.classList.add('hidden');
            modal.classList.remove('flex');
        }

        function handleTaskSubmit(e) {
            e.preventDefault();
            
            const id = document.getElementById('taskId').value;
            const title = document.getElementById('taskInputTitle').value;
            const description = document.getElementById('taskInputDesc').value;
            const column = document.getElementById('taskColumn').value;

            if (id) {
                // Update existing
                const index = tasks.findIndex(t => t.id === id);
                if (index > -1) {
                    tasks[index].title = title;
                    tasks[index].description = description;
                }
            } else {
                // Create new
                const newTask = {
                    id: 'task-' + Date.now(),
                    title,
                    description,
                    column
                };
                tasks.push(newTask);
            }

            saveData();
            renderBoard();
            closeModal('taskModal');
        }

        // Delete Logic
        function openDeleteModal(id) {
            taskToDeleteId = id;
            const modal = document.getElementById('deleteModal');
            modal.classList.remove('hidden');
            modal.classList.add('flex');
        }

        document.getElementById('confirmDeleteBtn').addEventListener('click', () => {
            if (taskToDeleteId) {
                tasks = tasks.filter(t => t.id !== taskToDeleteId);
                saveData();
                renderBoard();
                taskToDeleteId = null;
                closeModal('deleteModal');
            }
        });

        // Close modals on clicking outside
        window.onclick = function(event) {
            if (event.target.classList.contains('fixed')) { // Checks if clicking the overlay
                event.target.classList.add('hidden');
                event.target.classList.remove('flex');
            }
        }

        // Start App
        init();

    </script>
</body>
</html>
