    <!-- JavaScript -->
    <script>
        // Namespace for localStorage to avoid conflicts
        const STORAGE_NAMESPACE = 'kanbanApp_';
        
        // Global variables
        let draggedCard = null;
        let cardToDelete = null;
        let cardIdCounter = 1;
        
        // Initialize the application
        document.addEventListener('DOMContentLoaded', function() {
            loadBoardState();
            initializeDragAndDrop();
            updateCardCounts();
        });
        
        // Load board state from localStorage
        function loadBoardState() {
            const savedState = localStorage.getItem(STORAGE_NAMESPACE + 'boardState');
            
            if (savedState) {
                const state = JSON.parse(savedState);
                cardIdCounter = state.nextId || 1;
                
                // Load cards for each column
                Object.keys(state.cards || {}).forEach(column => {
                    const container = document.querySelector(`[data-column="${column}"] [data-cards-container]`);
                    if (container) {
                        state.cards[column].forEach(cardData => {
                            createCardElement(cardData, column);
                        });
                    }
                });
            } else {
                // Create dummy cards if no saved state
                createDummyCards();
            }
        }
        
        // Save board state to localStorage
        function saveBoardState() {
            const state = {
                nextId: cardIdCounter,
                cards: {}
            };
            
            ['backlog', 'inprogress', 'review', 'done'].forEach(column => {
                const container = document.querySelector(`[data-column="${column}"] [data-cards-container]`);
                const cards = Array.from(container.querySelectorAll('.card')).map(card => ({
                    id: card.dataset.cardId,
                    content: card.querySelector('.card-content').textContent.trim()
                }));
                state.cards[column] = cards;
            });
            
            localStorage.setItem(STORAGE_NAMESPACE + 'boardState', JSON.stringify(state));
        }
        
        // Create dummy cards for first-time users
        function createDummyCards() {
            const dummyCards = [
                { column: 'backlog', content: 'Design new landing page' },
                { column: 'backlog', content: 'Implement user authentication' },
                { column: 'inprogress', content: 'Fix responsive layout issues' },
                { column: 'review', content: 'Update documentation' },
                { column: 'done', content: 'Setup CI/CD pipeline' }
            ];
            
            dummyCards.forEach(card => {
                createCardElement({ id: cardIdCounter++, content: card.content }, card.column);
            });
            
            saveBoardState();
        }
        
        // Create a card element
        function createCardElement(cardData, column) {
            const card = document.createElement('div');
            card.className = 'card bg-white border border-gray-200 rounded-lg p-4 mb-3 shadow-sm hover:shadow-md fade-in';
            card.draggable = true;
            card.dataset.cardId = cardData.id;
            
            card.innerHTML = `
                <div class="card-content text-gray-800 mb-3">${escapeHtml(cardData.content)}</div>
                <div class="card-edit-form">
                    <textarea class="w-full p-2 border border-gray-300 rounded resize-none focus:outline-none focus:ring-2 focus:ring-indigo-500" rows="3">${escapeHtml(cardData.content)}</textarea>
                    <div class="flex justify-end space-x-2 mt-2">
                        <button onclick="cancelEdit(this)" class="px-3 py-1 bg-gray-200 text-gray-700 rounded hover:bg-gray-300 transition-colors text-sm">
                            Cancel
                        </button>
                        <button onclick="saveEdit(this)" class="px-3 py-1 bg-indigo-500 text-white rounded hover:bg-indigo-600 transition-colors text-sm">
                            Save
                        </button>
                    </div>
                </div>
                <div class="flex justify-between items-center">
                    <button onclick="editCard(this)" class="text-indigo-500 hover:text-indigo-700 transition-colors">
                        <i class="fas fa-edit"></i>
                    </button>
                    <button onclick="deleteCard(this)" class="text-red-500 hover:text-red-700 transition-colors">
                        <i class="fas fa-trash"></i>
                    </button>
                </div>
            `;
            
            const container = document.querySelector(`[data-column="${column}"] [data-cards-container]`);
            container.appendChild(card);
            
            // Add drag event listeners
            card.addEventListener('dragstart', handleDragStart);
            card.addEventListener('dragend', handleDragEnd);
        }
        
        // Add a new card to a column
        function addCard(column) {
            const cardData = {
                id: cardIdCounter++,
                content: 'New task card'
            };
            
            createCardElement(cardData, column);
            saveBoardState();
            updateCardCounts();
            
            // Automatically enter edit mode for new card
            const newCard = document.querySelector(`[data-card-id="${cardData.id}"]`);
            editCard(newCard.querySelector('[onclick*="editCard"]'));
        }
        
        // Edit card content in place
        function editCard(button) {
            const card = button.closest('.card');
            card.classList.add('editing');
            const textarea = card.querySelector('textarea');
            textarea.focus();
            textarea.select();
            
            // Auto-resize textarea
            textarea.style.height = 'auto';
            textarea.style.height = textarea.scrollHeight + 'px';
        }
        
        // Save edited card
        function saveEdit(button) {
            const card = button.closest('.card');
            const textarea = card.querySelector('textarea');
            const content = textarea.value.trim();
            
            if (content) {
                card.querySelector('.card-content').textContent = content;
                card.classList.remove('editing');
                saveBoardState();
            }
        }
        
        // Cancel editing
        function cancelEdit(button) {
            const card = button.closest('.card');
            const textarea = card.querySelector('textarea');
            const originalContent = card.querySelector('.card-content').textContent;
            textarea.value = originalContent;
            card.classList.remove('editing');
        }
        
        // Delete a card with confirmation
        function deleteCard(button) {
            const card = button.closest('.card');
            cardToDelete = card;
            document.getElementById('confirmModal').classList.add('show');
        }
        
        // Confirm card deletion
        document.getElementById('confirmDeleteBtn').addEventListener('click', function() {
            if (cardToDelete) {
                cardToDelete.remove();
                cardToDelete = null;
                saveBoardState();
                updateCardCounts();
            }
            closeConfirmModal();
        });
        
        // Close confirmation modal
        function closeConfirmModal() {
            document.getElementById('confirmModal').classList.remove('show');
            cardToDelete = null;
        }
        
        // Initialize drag and drop functionality
        function initializeDragAndDrop() {
            const columns = document.querySelectorAll('.column');
            
            columns.forEach(column => {
                const container = column.querySelector('[data-cards-container]');
                
                container.addEventListener('dragover', handleDragOver);
                container.addEventListener('drop', handleDrop);
                container.addEventListener('dragenter', handleDragEnter);
                container.addEventListener('dragleave', handleDragLeave);
            });
        }
        
        // Drag start handler
        function handleDragStart(e) {
            draggedCard = this;
            this.classList.add('dragging');
            e.dataTransfer.effectAllowed = 'move';
            e.dataTransfer.setData('text/html', this.innerHTML);
        }
        
        // Drag end handler
        function handleDragEnd(e) {
            this.classList.remove('dragging');
            
            // Remove drag-over class from all columns
            document.querySelectorAll('.column').forEach(column => {
                column.classList.remove('drag-over');
            });
        }
        
        // Drag over handler
        function handleDragOver(e) {
            if (e.preventDefault) {
                e.preventDefault();
            }
            e.dataTransfer.dropEffect = 'move';
            return false;
        }
        
        // Drag enter handler
        function handleDragEnter(e) {
            this.closest('.column').classList.add('drag-over');
        }
        
        // Drag leave handler
        function handleDragLeave(e) {
            if (e.target === this) {
                this.closest('.column').classList.remove('drag-over');
            }
        }
        
        // Drop handler
        function handleDrop(e) {
            if (e.stopPropagation) {
                e.stopPropagation();
            }
            
            if (draggedCard && draggedCard !== this) {
                const targetColumn = this.closest('.column');
                const targetContainer = this;
                
                // Append the dragged card to the new column
                targetContainer.appendChild(draggedCard);
                
                // Check if card was moved to "Done" column
                if (targetColumn.dataset.column === 'done') {
                    triggerConfetti();
                }
                
                saveBoardState();
                updateCardCounts();
            }
            
            targetColumn.classList.remove('drag-over');
            return false;
        }
        
        // Trigger confetti animation
        function triggerConfetti() {
            const colors = ['#ff6b6b', '#4ecdc4', '#45b7d1', '#96ceb4', '#feca57', '#ff8cc8', '#a8e6cf'];
            
            for (let i = 0; i < 50; i++) {
                setTimeout(() => {
                    const confetti = document.createElement('div');
                    confetti.className = 'confetti';
                    confetti.style.left = Math.random() * 100 + '%';
                    confetti.style.backgroundColor = colors[Math.floor(Math.random() * colors.length)];
                    confetti.style.animationDelay = Math.random() * 0.5 + 's';
                    confetti.style.animationDuration = (Math.random() * 2 + 2) + 's';
                    document.body.appendChild(confetti);
                    
                    // Remove confetti after animation
                    setTimeout(() => confetti.remove(), 3000);
                }, i * 30);
            }
        }
        
        // Update card counts for each column
        function updateCardCounts() {
            ['backlog', 'inprogress', 'review', 'done'].forEach(column => {
                const container = document.querySelector(`[data-column="${column}"] [data-cards-container]`);
                const count = container.querySelectorAll('.card').length;
                const countElement = document.querySelector(`[data-column="${column}"] .card-count`);
                countElement.textContent = count;
            });
        }
        
        // Clear all data
        function clearAllData() {
            if (confirm('Are you sure you want to clear all data? This will remove all cards and cannot be undone.')) {
                localStorage.removeItem(STORAGE_NAMESPACE + 'boardState');
                location.reload();
            }
        }
        
        // Escape HTML to prevent XSS
        function escapeHtml(text) {
            const map = {
                '&': '&amp;',
                '<': '&lt;',
                '>': '&gt;',
                '"': '&quot;',
                "'": '&#039;'
            };
            return text.replace(/[&<>"']/g, m => map[m]);
        }
        
        // Handle keyboard shortcuts
        document.addEventListener('keydown', function(e) {
            // Escape key to close modal
            if (e.key === 'Escape') {
                closeConfirmModal();
                
                // Cancel any ongoing edits
                document.querySelectorAll('.card.editing').forEach(card => {
                    card.classList.remove('editing');
                });
            }
            
            // Ctrl/Cmd + S to save
            if ((e.ctrlKey || e.metaKey) && e.key === 's') {
                e.preventDefault();
                saveBoardState();
            }
        });
    </script>
