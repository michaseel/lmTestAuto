  <style>

<body class="bg-gradient-to-br from-indigo-50 to-purple-50 min-h-screen">
  <div class="container mx-auto px-4 py-8">
    <h1 class="text-4xl font-bold text-center mb-2 text-indigo-800">Kanban Board</h1>
    <p class="text-center mb-8 text-gray-600">Drag and drop cards between columns to track your tasks</p>
    
    <!-- Kanban board container -->
    <div id="kanban-board" class="grid grid-cols-1 md:grid-cols-2 lg:grid-cols-4 gap-6">
      <!-- Columns will be dynamically generated here -->
    </div>
    
    <!-- Modal for confirmation -->
    <div id="confirmation-modal" class="fixed inset-0 flex items-center justify-center hidden modal-backdrop">
      <div class="modal-content bg-white rounded-lg p-6 max-w-md w-full mx-4 shadow-xl">
        <h3 class="text-xl font-semibold text-gray-900 mb-4">Confirm Deletion</h3>
        <p class="text-gray-700 mb-6">Are you sure you want to delete this card? This action cannot be undone.</p>
        <div class="flex justify-end space-x-4">
          <button id="cancel-delete" class="px-4 py-2 bg-gray-200 text-gray-800 rounded hover:bg-gray-300 transition-colors">
            Cancel
          </button>
          <button id="confirm-delete" class="px-4 py-2 bg-red-600 text-white rounded hover:bg-red-700 transition-colors">
            Delete

    <!-- Modal for adding/editing cards -->
    <div id="card-modal" class="fixed inset-0 flex items-center justify-center hidden modal-backdrop">
      <div class="modal-content bg-white rounded-lg p-6 max-w-md w-full mx-4 shadow-xl">
        <h3 id="card-modal-title" class="text-xl font-semibold text-gray-900 mb-4">Add New Card</h3>
        <textarea id="card-content" class="w-full border border-gray-300 rounded-lg p-3 mb-4 h-32 focus:outline-none focus:ring-2 focus:ring-indigo-500" placeholder="Enter card content..."></textarea>
        <div class="flex justify-end space-x-4">
          <button id="cancel-card" class="px-4 py-2 bg-gray-200 text-gray-800 rounded hover:bg-gray-300 transition-colors">
            Cancel
          </button>

            Save

  <script>
    // Main application code
    document.addEventListener('DOMContentLoaded', function() {
      // Constants for column names and localStorage key
      const COLUMNS = ['Backlog', 'In Progress', 'Review', 'Done'];
      const STORAGE_KEY = 'my_kanban_board_data';
      
      // Column colors for visual distinction
      const COLUMN_COLORS = {
        'Backlog': 'bg-blue-50 border-blue-200',
        'In Progress': 'bg-yellow-50 border-yellow-200',
        'Review': 'bg-purple-50 border-purple-200',
        'Done': 'bg-green-50 border-green-200'
      };
      
      // Card header colors
      const CARD_COLORS = {
        'Backlog': 'bg-blue-100',
        'In Progress': 'bg-yellow-100',
        'Review': 'bg-purple-100',
        'Done': 'bg-green-100'
      };
      
      // Track the card being dragged
      let draggedCard = null;
      
      // Track card being edited or deleted
      let currentCardId = null;
      let currentColumn = null;
      let isEditing = false;
      
      // Initialize the board
      initializeBoard();
      
      /**
       * Initialize the Kanban board by creating columns and loading data
       */
      function initializeBoard() {
        const board = document.getElementById('kanban-board');
        
        // Create columns
        COLUMNS.forEach(columnName => {
          const columnElement = createColumn(columnName);
          board.appendChild(columnElement);
        });
        
        // Load data from localStorage or use default data
        loadBoardData();
        
        // Set up event listeners for modals
        setupModalListeners();
      }
      
      /**
       * Create a column element with the given name
       * @param {string} columnName - The name of the column
       * @return {HTMLElement} The created column element
       */
      function createColumn(columnName) {
        const column = document.createElement('div');
        column.className = `column-transition ${COLUMN_COLORS[columnName]} p-4 rounded-lg border shadow-md flex flex-col h-[calc(100vh-12rem)] min-h-[500px]`;
        column.dataset.column = columnName;
        
        // Column header
        const header = document.createElement('div');
        header.className = 'flex justify-between items-center mb-4';
        
        const title = document.createElement('h2');
        title.className = 'text-xl font-bold text-gray-800';
        title.textContent = columnName;
        
        const count = document.createElement('span');
        count.className = 'card-count px-2 py-1 bg-white rounded-full text-sm font-semibold text-gray-600 shadow-sm';
        count.textContent = '0';
        
        header.appendChild(title);
        header.appendChild(count);
        
        // Cards container (scrollable)
        const cardsContainer = document.createElement('div');
        cardsContainer.className = 'cards-container flex-grow overflow-y-auto mb-4 space-y-3';
        cardsContainer.dataset.column = columnName;
        
        // Add card button
        const addButton = document.createElement('button');
        addButton.className = 'w-full py-2 bg-white rounded-lg border border-gray-200 text-gray-600 hover:bg-gray-50 hover:text-gray-800 transition-colors flex items-center justify-center space-x-2 shadow-sm';
        addButton.innerHTML = '<svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" viewBox="0 0 20 20" fill="currentColor"><path fill-rule="evenodd" d="M10 5a1 1 0 011 1v3h3a1 1 0 110 2h-3v3a1 1 0 11-2 0v-3H6a1 1 0 110-2h3V6a1 1 0 011-1z" clip-rule="evenodd" /></svg><span>Add Card</span>';
        
        // Add event listener to the add button
        addButton.addEventListener('click', () => openCardModal(columnName));
        
        // Set up drag and drop event listeners for the column
        cardsContainer.addEventListener('dragover', handleDragOver);
        cardsContainer.addEventListener('dragleave', handleDragLeave);
        cardsContainer.addEventListener('drop', handleDrop);
        
        column.appendChild(header);
        column.appendChild(cardsContainer);
        column.appendChild(addButton);
        
        return column;
      }
      
      /**
       * Create a card element with the given data
       * @param {string} id - Unique identifier for the card
       * @param {string} content - The content of the card
       * @param {string} column - The column the card belongs to
       * @return {HTMLElement} The created card element
       */
      function createCard(id, content, column) {
        const card = document.createElement('div');
        card.className = `card card-transition bg-white rounded-lg shadow overflow-hidden`;
        card.draggable = true;
        card.dataset.id = id;
        
        // Card header with the column color
        const cardHeader = document.createElement('div');
        cardHeader.className = `${CARD_COLORS[column]} px-3 py-2 flex justify-between items-center`;
        
        const cardTitle = document.createElement('div');
        cardTitle.className = 'text-xs font-semibold text-gray-700';
        cardTitle.textContent = `#${id.substring(0, 8)}`;
        
        const actionButtons = document.createElement('div');
        actionButtons.className = 'flex space-x-2';
        
        // Edit button
        const editButton = document.createElement('button');
        editButton.className = 'text-gray-600 hover:text-gray-900';
        editButton.innerHTML = '<svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M15.232 5.232l3.536 3.536m-2.036-5.036a2.5 2.5 0 113.536 3.536L6.5 21.036H3v-3.572L16.732 3.732z" /></svg>';
        editButton.addEventListener('click', (e) => {
          e.stopPropagation();
          openCardModal(column, id, content);
        });
        
        // Delete button
        const deleteButton = document.createElement('button');
        deleteButton.className = 'text-gray-600 hover:text-red-600';
        deleteButton.innerHTML = '<svg xmlns="http://www.w3.org/2000/svg" class="h-4 w-4" fill="none" viewBox="0 0 24 24" stroke="currentColor"><path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16" /></svg>';
        deleteButton.addEventListener('click', (e) => {
          e.stopPropagation();
          openDeleteConfirmation(id, column);
        });
        
        actionButtons.appendChild(editButton);
        actionButtons.appendChild(deleteButton);
        
        cardHeader.appendChild(cardTitle);
        cardHeader.appendChild(actionButtons);
        
        // Card content
        const cardContent = document.createElement('div');
        cardContent.className = 'p-4 text-gray-700';
        cardContent.textContent = content;
        
        card.appendChild(cardHeader);
        card.appendChild(cardContent);
        
        // Set up drag event listeners
        card.addEventListener('dragstart', handleDragStart);
        card.addEventListener('dragend', handleDragEnd);
        
        return card;
      }
      
      /**
       * Load board data from localStorage or use default data
       */
      function loadBoardData() {
        let boardData = JSON.parse(localStorage.getItem(STORAGE_KEY));
        
        // If no data exists, create sample data
        if (!boardData) {
          boardData = {
            'Backlog': [
              { id: generateId(), content: 'Research user requirements for new feature' },
              { id: generateId(), content: 'Create wireframes for homepage redesign' }
            ],
            'In Progress': [
              { id: generateId(), content: 'Implement authentication system' }
            ],
            'Review': [
              { id: generateId(), content: 'Code review for payment integration PR' }
            ],
            'Done': [
              { id: generateId(), content: 'Set up CI/CD pipeline' }
            ]
          };
          
          // Save default data to localStorage
          saveBoardData(boardData);
        }
        
        // Render cards for each column
        COLUMNS.forEach(columnName => {
          const cardsContainer = document.querySelector(`.cards-container[data-column="${columnName}"]`);
          
          // Clear existing cards
          cardsContainer.innerHTML = '';
          
          // Add cards from data
          if (boardData[columnName]) {
            boardData[columnName].forEach(cardData => {
              const card = createCard(cardData.id, cardData.content, columnName);
              cardsContainer.appendChild(card);
            });
          }
          
          // Update card count
          updateCardCount(columnName);
        });
      }
      
      /**
       * Save board data to localStorage
       * @param {Object} data - The board data to save
       */
      function saveBoardData(data) {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(data));
      }
      
      /**
       * Get current board data from the DOM
       * @return {Object} The current board data
       */
      function getBoardDataFromDOM() {
        const data = {};
        
        COLUMNS.forEach(columnName => {
          const cardsContainer = document.querySelector(`.cards-container[data-column="${columnName}"]`);
          const cards = cardsContainer.querySelectorAll('.card');
          
          data[columnName] = Array.from(cards).map(card => {
            return {
              id: card.dataset.id,
              content: card.querySelector('.p-4').textContent
            };
          });
        });
        
        return data;
      }
      
      /**
       * Generate a unique ID for a card
       * @return {string} A unique ID
       */
      function generateId() {
        return Date.now().toString(36) + Math.random().toString(36).substr(2, 5);
      }
      
      /**
       * Update the card count for a specific column
       * @param {string} columnName - The name of the column
       */
      function updateCardCount(columnName) {
        const column = document.querySelector(`.column-transition[data-column="${columnName}"]`);
        const count = column.querySelector('.card-count');
        const cardsContainer = column.querySelector('.cards-container');
        const cardCount = cardsContainer.querySelectorAll('.card').length;
        
        count.textContent = cardCount.toString();
      }
      
      /**
       * Set up event listeners for modals
       */
      function setupModalListeners() {
        // Delete confirmation modal
        const confirmationModal = document.getElementById('confirmation-modal');
        const cancelDeleteBtn = document.getElementById('cancel-delete');
        const confirmDeleteBtn = document.getElementById('confirm-delete');
        
        cancelDeleteBtn.addEventListener('click', closeDeleteConfirmation);
        confirmDeleteBtn.addEventListener('click', deleteCard);
        
        // Card add/edit modal
        const cardModal = document.getElementById('card-modal');
        const cancelCardBtn = document.getElementById('cancel-card');
        const saveCardBtn = document.getElementById('save-card');
        
        cancelCardBtn.addEventListener('click', closeCardModal);
        saveCardBtn.addEventListener('click', saveCard);
      }
      
      /**
       * Open the delete confirmation modal
       * @param {string} cardId - The ID of the card to delete
       * @param {string} column - The column the card belongs to
       */
      function openDeleteConfirmation(cardId, column) {
        const modal = document.getElementById('confirmation-modal');
        currentCardId = cardId;
        currentColumn = column;
        
        modal.classList.remove('hidden');
        setTimeout(() => modal.classList.add('modal-show'), 10);
      }
      
      /**
       * Close the delete confirmation modal
       */
      function closeDeleteConfirmation() {
        const modal = document.getElementById('confirmation-modal');
        modal.classList.remove('modal-show');
        setTimeout(() => {
          modal.classList.add('hidden');
          currentCardId = null;
          currentColumn = null;
        }, 300);
      }
      
      /**
       * Delete the current card
       */
      function deleteCard() {
        if (currentCardId && currentColumn) {
          const cardsContainer = document.querySelector(`.cards-container[data-column="${currentColumn}"]`);
          const card = cardsContainer.querySelector(`.card[data-id="${currentCardId}"]`);
          
          if (card) {
            card.remove();
            
            // Update card count
            updateCardCount(currentColumn);
            
            // Save updated data
            const boardData = getBoardDataFromDOM();
            saveBoardData(boardData);
          }
        }
        
        closeDeleteConfirmation();
      }
      
      /**
       * Open the card modal for adding or editing a card
       * @param {string} column - The column the card belongs to
       * @param {string} [cardId] - The ID of the card to edit (if editing)
       * @param {string} [content] - The content of the card (if editing)
       */
      function openCardModal(column, cardId, content) {
        const modal = document.getElementById('card-modal');
        const modalTitle = document.getElementById('card-modal-title');
        const contentTextarea = document.getElementById('card-content');
        
        currentColumn = column;
        currentCardId = cardId;
        isEditing = !!cardId;
        
        modalTitle.textContent = isEditing ? 'Edit Card' : 'Add New Card';
        contentTextarea.value = content || '';
        
        modal.classList.remove('hidden');
        setTimeout(() => {
          modal.classList.add('modal-show');
          contentTextarea.focus();
        }, 10);
      }
      
      /**
       * Close the card modal
       */
      function closeCardModal() {
        const modal = document.getElementById('card-modal');
        modal.classList.remove('modal-show');
        setTimeout(() => {
          modal.classList.add('hidden');
          currentCardId = null;
          currentColumn = null;
          isEditing = false;
        }, 300);
      }
      
      /**
       * Save the current card (add new or update existing)
       */
      function saveCard() {
        const contentTextarea = document.getElementById('card-content');
        const content = contentTextarea.value.trim();
        
        if (content && currentColumn) {
          const cardsContainer = document.querySelector(`.cards-container[data-column="${currentColumn}"]`);
          
          if (isEditing && currentCardId) {
            // Update existing card
            const card = cardsContainer.querySelector(`.card[data-id="${currentCardId}"]`);
            if (card) {
              card.querySelector('.p-4').textContent = content;
            }
          } else {
            // Add new card
            const newCardId = generateId();
            const newCard = createCard(newCardId, content, currentColumn);
            cardsContainer.appendChild(newCard);
          }
          
          // Update card count
          updateCardCount(currentColumn);
          
          // Save updated data
          const boardData = getBoardDataFromDOM();
          saveBoardData(boardData);
        }
        
        closeCardModal();
      }
      
      /**
       * Handle the start of a drag operation
       * @param {DragEvent} e - The drag event
       */
      function handleDragStart(e) {
        draggedCard = e.target;
        e.dataTransfer.effectAllowed = 'move';
        e.dataTransfer.setData('text/plain', ''); // Required for Firefox
        
        // Add dragging class
        setTimeout(() => {
          draggedCard.classList.add('opacity-50');
        }, 0);
      }
      
      /**
       * Handle the end of a drag operation
       */
      function handleDragEnd() {
        if (draggedCard) {
          draggedCard.classList.remove('opacity-50');
          draggedCard = null;
        }
        
        // Remove drag-over class from all columns
        document.querySelectorAll('.cards-container').forEach(container => {
          container.classList.remove('drag-over');
        });
      }
      
      /**
       * Handle the dragover event
       * @param {DragEvent} e - The drag event
       */
      function handleDragOver(e) {
        if (e.preventDefault) {
          e.preventDefault(); // Allows dropping
        }
        
        e.dataTransfer.dropEffect = 'move';
        e.currentTarget.classList.add('drag-over');
        return false;
      }
      
      /**
       * Handle the dragleave event
       * @param {DragEvent} e - The drag event
       */
      function handleDragLeave(e) {
        e.currentTarget.classList.remove('drag-over');
      }
      
      /**
       * Handle the drop event
       * @param {DragEvent} e - The drag event
       */
      function handleDrop(e) {
        if (e.stopPropagation) {
          e.stopPropagation(); // Stops browser from redirecting
        }
        
        e.currentTarget.classList.remove('drag-over');
        
        if (draggedCard) {
          const sourceColumn = draggedCard.closest('.cards-container').dataset.column;
          const targetColumn = e.currentTarget.dataset.column;
          
          // Only proceed if dropping into a different column
          if (sourceColumn !== targetColumn) {
            // Clone the card to avoid issues with removing during drag
            const cardClone = draggedCard.cloneNode(true);
            
            // Re-attach event listeners to the clone
            cardClone.addEventListener('dragstart', handleDragStart);
            cardClone.addEventListener('dragend', handleDragEnd);
            
            const editButton = cardClone.querySelector('button:first-child');
            const deleteButton = cardClone.querySelector('button:last-child');
            
            const cardId = cardClone.dataset.id;
            const content = cardClone.querySelector('.p-4').textContent;
            
            editButton.addEventListener('click', (e) => {
              e.stopPropagation();
              openCardModal(targetColumn, cardId, content);
            });
            
            deleteButton.addEventListener('click', (e) => {
              e.stopPropagation();
              openDeleteConfirmation(cardId, targetColumn);
            });
            
            // Update the card header color based on the new column
            const cardHeader = cardClone.querySelector('div:first-child');
            cardHeader.className = `${CARD_COLORS[targetColumn]} px-3 py-2 flex justify-between items-center`;
            
            // Remove the original card
            draggedCard.remove();
            
            // Add the card to the target column
            e.currentTarget.appendChild(cardClone);
            
            // Update card counts
            updateCardCount(sourceColumn);
            updateCardCount(targetColumn);
            
            // Save the updated board state
            const boardData = getBoardDataFromDOM();
            saveBoardData(boardData);
            
            // Trigger confetti if moving to Done column
            if (targetColumn === 'Done') {
              triggerConfetti();
            }
          }
        }
        
        return false;
      }
      
      /**
       * Trigger a confetti animation
       */
      function triggerConfetti() {
        confetti({
          particleCount: 100,
          spread: 70,
          origin: { y: 0.6 }
        });
      }
    });
  </script>
</body>
</html>
