    <script>
        // Kanban Board Application
        (function() {
            'use strict';
            
            // Local storage namespace
            const STORAGE_KEY = 'kanban-board-data';
            
            // State management
            let kanbanState = {
                cards: [],
                nextId: 1,
                currentEditId: null,
                currentDeleteId: null,
                dragSource: null
            };
            
            // Column definitions
            const COLUMNS = [
                { id: 'backlog', name: 'Backlog', color: 'gray' },
                { id: 'in-progress', name: 'In Progress', color: 'blue' },
                { id: 'review', name: 'Review', color: 'amber' },
                { id: 'done', name: 'Done', color: 'emerald' }
            ];
            
            // DOM Elements
            const elements = {
                columns: document.querySelectorAll('.kanban-column'),
                cardContainers: document.querySelectorAll('.cards-container'),
                addCardButtons: document.querySelectorAll('.add-card-btn'),
                editModal: document.getElementById('edit-modal'),
                deleteModal: document.getElementById('delete-modal'),
                editTitle: document.getElementById('edit-title'),
                editContent: document.getElementById('edit-content'),
                saveEditBtn: document.getElementById('save-edit'),
                cancelEditBtn: document.getElementById('cancel-edit'),
                confirmDeleteBtn: document.getElementById('confirm-delete'),
                cancelDeleteBtn: document.getElementById('cancel-delete'),
                clearStorageBtn: document.getElementById('clear-storage'),
                addDummyBtn: document.getElementById('add-dummy'),
                cardCount: document.getElementById('card-count')
            };
            
            // Initialize the application
            function init() {
                loadState();
                renderBoard();
                setupEventListeners();
                
                // If no cards exist, create some dummy cards
                if (kanbanState.cards.length === 0) {
                    createDummyCards();
                }
            }
            
            // Load state from localStorage
            function loadState() {
                const savedState = localStorage.getItem(STORAGE_KEY);
                if (savedState) {
                    try {
                        const parsed = JSON.parse(savedState);
                        kanbanState.cards = parsed.cards || [];
                        kanbanState.nextId = parsed.nextId || 1;
                    } catch (e) {
                        console.error('Failed to load saved state:', e);
                        kanbanState.cards = [];
                        kanbanState.nextId = 1;
                    }
                }
            }
            
            // Save state to localStorage
            function saveState() {
                localStorage.setItem(STORAGE_KEY, JSON.stringify({
                    cards: kanbanState.cards,
                    nextId: kanbanState.nextId
                }));
            }
            
            // Create initial dummy cards
            function createDummyCards() {
                const dummyCards = [
                    { id: kanbanState.nextId++, title: 'Design Homepage', content: 'Create wireframes and mockups for the homepage', column: 'backlog', createdAt: new Date().toISOString() },
                    { id: kanbanState.nextId++, title: 'Implement Login', content: 'Build the login functionality with validation', column: 'in-progress', createdAt: new Date().toISOString() },
                    { id: kanbanState.nextId++, title: 'Fix Mobile Layout', content: 'Responsive design adjustments for mobile devices', column: 'review', createdAt: new Date().toISOString() },
                    { id: kanbanState.nextId++, title: 'Setup CI/CD', content: 'Configure continuous integration and deployment pipeline', column: 'done', createdAt: new Date().toISOString() },
                    { id: kanbanState.nextId++, title: 'Write Documentation', content: 'Document the API endpoints and usage', column: 'backlog', createdAt: new Date().toISOString() },
                    { id: kanbanState.nextId++, title: 'Optimize Images', content: 'Compress and optimize images for faster loading', column: 'in-progress', createdAt: new Date().toISOString() }
                ];
                
                kanbanState.cards = dummyCards;
                saveState();
                renderBoard();
            }
            
            // Render the entire board
            function renderBoard() {
                // Clear all columns
                elements.cardContainers.forEach(container => {
                    container.innerHTML = '';
                });
                
                // Sort cards by creation date (oldest first)
                kanbanState.cards.sort((a, b) => new Date(a.createdAt) - new Date(b.createdAt));
                
                // Add cards to their respective columns
                kanbanState.cards.forEach(card => {
                    const cardElement = createCardElement(card);
                    const container = document.querySelector(`.cards-container[data-column="${card.column}"]`);
                    container.appendChild(cardElement);
                });
                
                // Update column counts
                updateColumnCounts();
                
                // Update total card count
                elements.cardCount.textContent = kanbanState.cards.length;
            }
            
            // Create a card element
            function createCardElement(card) {
                const cardEl = document.createElement('div');
                cardEl.className = 'card bg-white rounded-lg shadow p-4 mb-3 cursor-move border-l-4';
                cardEl.dataset.id = card.id;
                cardEl.dataset.column = card.column;
                cardEl.draggable = true;
                
                // Set border color based on column
                const columnInfo = COLUMNS.find(col => col.id === card.column);
                if (columnInfo) {
                    cardEl.classList.add(`border-${columnInfo.color}-500`);
                }
                
                // Add card content
                cardEl.innerHTML = `
                    <div class="flex justify-between items-start mb-2">
                        <h3 class="font-bold text-gray-800 card-title">${escapeHtml(card.title)}</h3>
                        <button class="delete-btn text-gray-400 hover:text-red-500 transition-colors p-1 rounded">
                            <svg class="w-5 h-5" fill="none" stroke="currentColor" viewBox="0 0 24 24" xmlns="http://www.w3.org/2000/svg">
                                <path stroke-linecap="round" stroke-linejoin="round" stroke-width="2" d="M19 7l-.867 12.142A2 2 0 0116.138 21H7.862a2 2 0 01-1.995-1.858L5 7m5 4v6m4-6v6m1-10V4a1 1 0 00-1-1h-4a1 1 0 00-1 1v3M4 7h16"></path>
                            </svg>
                        </button>
                    </div>
                    <p class="text-gray-600 text-sm card-content">${escapeHtml(card.content)}</p>
                    <div class="mt-3 text-xs text-gray-400">
                        Created: ${formatDate(card.createdAt)}
                    </div>
                `;
                
                // Add event listeners for this card
                cardEl.addEventListener('dragstart', handleDragStart);
                cardEl.addEventListener('dragend', handleDragEnd);
                cardEl.addEventListener('dblclick', () => openEditModal(card.id));
                
                // Add delete button listener
                const deleteBtn = cardEl.querySelector('.delete-btn');
                deleteBtn.addEventListener('click', (e) => {
                    e.stopPropagation(); // Prevent triggering card click events
                    openDeleteModal(card.id);
                });
                
                return cardEl;
            }
            
            // Update column card counts
            function updateColumnCounts() {
                COLUMNS.forEach(column => {
                    const count = kanbanState.cards.filter(card => card.column === column.id).length;
                    const countElement = document.querySelector(`.column-count[data-column="${column.id}"]`);
                    if (countElement) {
                        countElement.textContent = count;
                    }
                });
            }
            
            // Drag and drop handlers
            function handleDragStart(e) {
                kanbanState.dragSource = {
                    id: parseInt(e.target.dataset.id),
                    column: e.target.dataset.column
                };
                
                e.target.classList.add('dragging');
                e.dataTransfer.setData('text/plain', e.target.dataset.id);
                e.dataTransfer.effectAllowed = 'move';
            }
            
            function handleDragEnd(e) {
                e.target.classList.remove('dragging');
                
                // Remove drag-over class from all columns
                elements.columns.forEach(col => {
                    col.querySelector('.cards-container').classList.remove('drag-over');
                });
            }
            
            // Setup event listeners for drag and drop
            function setupDragAndDrop() {
                // Add dragover and drop events to each column's card container
                elements.cardContainers.forEach(container => {
                    container.addEventListener('dragover', (e) => {
                        e.preventDefault();
                        container.classList.add('drag-over');
                        e.dataTransfer.dropEffect = 'move';
                    });
                    
                    container.addEventListener('dragleave', () => {
                        container.classList.remove('drag-over');
                    });
                    
                    container.addEventListener('drop', (e) => {
                        e.preventDefault();
                        container.classList.remove('drag-over');
                        
                        const cardId = parseInt(e.dataTransfer.getData('text/plain'));
                        const targetColumn = container.dataset.column;
                        
                        if (cardId && kanbanState.dragSource) {
                            moveCardToColumn(cardId, targetColumn);
                        }
                    });
                });
            }
            
            // Move a card to a different column
            function moveCardToColumn(cardId, targetColumn) {
                const cardIndex = kanbanState.cards.findIndex(card => card.id === cardId);
                
                if (cardIndex !== -1) {
                    const previousColumn = kanbanState.cards[cardIndex].column;
                    kanbanState.cards[cardIndex].column = targetColumn;
                    
                    // Trigger confetti if moved to Done column
                    if (targetColumn === 'done' && previousColumn !== 'done') {
                        triggerConfetti();
                    }
                    
                    saveState();
                    renderBoard();
                }
            }
            
            // Trigger confetti animation
            function triggerConfetti() {
                confetti({
                    particleCount: 150,
                    spread: 70,
                    origin: { y: 0.6 }
                });
                
                // Add a second burst after a short delay
                setTimeout(() => {
                    confetti({
                        particleCount: 100,
                        angle: 60,
                        spread: 55,
                        origin: { x: 0 }
                    });
                }, 250);
                
                setTimeout(() => {
                    confetti({
                        particleCount: 100,
                        angle: 120,
                        spread: 55,
                        origin: { x: 1 }
                    });
                }, 400);
            }
            
            // Open edit modal
            function openEditModal(cardId) {
                const card = kanbanState.cards.find(card => card.id === cardId);
                if (!card) return;
                
                kanbanState.currentEditId = cardId;
                elements.editTitle.value = card.title;
                elements.editContent.value = card.content;
                elements.editModal.classList.remove('hidden');
            }
            
            // Save edited card
            function saveEditedCard() {
                const cardId = kanbanState.currentEditId;
                const cardIndex = kanbanState.cards.findIndex(card => card.id === cardId);
                
                if (cardIndex !== -1) {
                    const title = elements.editTitle.value.trim();
                    const content = elements.editContent.value.trim();
                    
                    if (title) {
                        kanbanState.cards[cardIndex].title = title;
                        kanbanState.cards[cardIndex].content = content;
                        saveState();
                        renderBoard();
                    }
                }
                
                closeEditModal();
            }
            
            // Close edit modal
            function closeEditModal() {
                elements.editModal.classList.add('hidden');
                kanbanState.currentEditId = null;
            }
            
            // Open delete confirmation modal
            function openDeleteModal(cardId) {
                kanbanState.currentDeleteId = cardId;
                elements.deleteModal.classList.remove('hidden');
            }
            
            // Delete card after confirmation
            function deleteCard() {
                const cardId = kanbanState.currentDeleteId;
                kanbanState.cards = kanbanState.cards.filter(card => card.id !== cardId);
                saveState();
                renderBoard();
                closeDeleteModal();
            }
            
            // Close delete modal
            function closeDeleteModal() {
                elements.deleteModal.classList.add('hidden');
                kanbanState.currentDeleteId = null;
            }
            
            // Add a new card to a column
            function addNewCard(columnId) {
                const newCard = {
                    id: kanbanState.nextId++,
                    title: 'New Task',
                    content: 'Describe your task here...',
                    column: columnId,
                    createdAt: new Date().toISOString()
                };
                
                kanbanState.cards.push(newCard);
                saveState();
                renderBoard();
                
                // Open edit modal for the new card
                setTimeout(() => {
                    openEditModal(newCard.id);
                }, 10);
            }
            
            // Clear all data from localStorage
            function clearAllData() {
                if (confirm('Are you sure you want to clear all data? This will remove all cards and cannot be undone.')) {
                    localStorage.removeItem(STORAGE_KEY);
                    kanbanState.cards = [];
                    kanbanState.nextId = 1;
                    renderBoard();
                }
            }
            
            // Setup all event listeners
            function setupEventListeners() {
                // Add card buttons
                elements.addCardButtons.forEach(button => {
                    button.addEventListener('click', () => {
                        const column = button.dataset.column;
                        addNewCard(column);
                    });
                });
                
                // Edit modal buttons
                elements.saveEditBtn.addEventListener('click', saveEditedCard);
                elements.cancelEditBtn.addEventListener('click', closeEditModal);
                
                // Delete modal buttons
                elements.confirmDeleteBtn.addEventListener('click', deleteCard);
                elements.cancelDeleteBtn.addEventListener('click', closeDeleteModal);
                
                // Close modals when clicking outside
                elements.editModal.addEventListener('click', (e) => {
                    if (e.target === elements.editModal) {
                        closeEditModal();
                    }
                });
                
                elements.deleteModal.addEventListener('click', (e) => {
                    if (e.target === elements.deleteModal) {
                        closeDeleteModal();
                    }
                });
                
                // Clear storage button
                elements.clearStorageBtn.addEventListener('click', clearAllData);
                
                // Add dummy cards button
                elements.addDummyBtn.addEventListener('click', createDummyCards);
                
                // Setup drag and drop
                setupDragAndDrop();
                
                // Allow Enter key to save in edit modal
                elements.editTitle.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter' && !e.shiftKey) {
                        e.preventDefault();
                        elements.editContent.focus();
                    }
                });
                
                elements.editContent.addEventListener('keydown', (e) => {
                    if (e.key === 'Enter' && e.ctrlKey) {
                        e.preventDefault();
                        saveEditedCard();
                    }
                });
            }
