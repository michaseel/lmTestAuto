            <h2 class="text-2xl font-bold text-slate-800 mb-4">üóëÔ∏è Delete Card?</h2>

    <script>
        /**
         * Kanban Board Application
         * A fully functional kanban board with drag-and-drop, localStorage persistence, and confetti animations
         */

        // Configuration
        const STORAGE_KEY = 'kanban_board_v1_data'; // Namespaced localStorage key
        const COLUMNS = [
            { id: 'backlog', title: 'Backlog', icon: 'üìã', color: 'column-backlog' },
            { id: 'progress', title: 'In Progress', icon: 'üöÄ', color: 'column-progress' },
            { id: 'review', title: 'Review', icon: 'üëÄ', color: 'column-review' },
            { id: 'done', title: 'Done', icon: '‚úÖ', color: 'column-done' }
        ];

        // Dummy data for initial load
        const DUMMY_CARDS = [
            { id: '1', content: 'Design new landing page', columnId: 'backlog' },
            { id: '2', content: 'Implement user authentication', columnId: 'progress' },
            { id: '3', content: 'Write API documentation', columnId: 'backlog' },
            { id: '4', content: 'Code review for payment module', columnId: 'review' },
            { id: '5', content: 'Setup CI/CD pipeline', columnId: 'done' },
            { id: '6', content: 'Database optimization', columnId: 'progress' }
        ];

        // Application state
        let cards = [];
        let draggedCard = null;
        let currentEditingCard = null;
        let currentDeletingCard = null;
        let currentModalColumn = null;

        /**
         * Initialize the application
         */
        function init() {
            loadCards();
            renderBoard();
            setupEventListeners();
        }

        /**
         * Load cards from localStorage or use dummy data
         */
        function loadCards() {
            const stored = localStorage.getItem(STORAGE_KEY);
            if (stored) {
                cards = JSON.parse(stored);
            } else {
                // No cards in storage, use dummy data
                cards = DUMMY_CARDS;
                saveCards();
            }
        }

        /**
         * Save cards to localStorage
         */
        function saveCards() {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(cards));
        }

        /**
         * Generate unique ID for new cards
         */
        function generateId() {
            return Date.now().toString(36) + Math.random().toString(36).substr(2);
        }

        /**
         * Render the entire kanban board
         */
        function renderBoard() {
            const board = document.getElementById('kanban-board');
            board.innerHTML = '';

            COLUMNS.forEach(column => {
                const columnEl = createColumnElement(column);
                board.appendChild(columnEl);
            });
        }

        /**
         * Create a column element
         */
        function createColumnElement(column) {
            const columnDiv = document.createElement('div');
            columnDiv.className = 'bg-white rounded-2xl shadow-lg overflow-hidden flex flex-col';
            columnDiv.dataset.columnId = column.id;

            // Column header with gradient
            const header = document.createElement('div');
            header.className = `${column.color} text-white p-4 flex items-center justify-between`;
            header.innerHTML = `
                <div class="flex items-center gap-2">
                    <span class="text-2xl">${column.icon}</span>
                    <h2 class="text-xl font-bold">${column.title}</h2>
                    <span class="bg-white bg-opacity-30 rounded-full px-3 py-1 text-sm font-semibold">
                        ${cards.filter(c => c.columnId === column.id).length}
                    </span>
                </div>
            `;

            // Add card button
            const addBtn = document.createElement('button');
            addBtn.className = 'bg-white bg-opacity-30 hover:bg-opacity-50 rounded-lg p-2 transition-all transform hover:scale-110';
            addBtn.innerHTML = '<span class="text-2xl">+</span>';
            addBtn.onclick = () => openAddCardModal(column.id);
            header.appendChild(addBtn);

            // Cards container with drag-and-drop
            const cardsContainer = document.createElement('div');
            cardsContainer.className = 'flex-1 p-4 space-y-3 overflow-y-auto min-h-[200px] max-h-[600px]';
            cardsContainer.dataset.columnId = column.id;
            
            // Add drag and drop event listeners
            cardsContainer.addEventListener('dragover', handleDragOver);
            cardsContainer.addEventListener('drop', handleDrop);
            cardsContainer.addEventListener('dragleave', handleDragLeave);

            // Render cards for this column
            const columnCards = cards.filter(card => card.columnId === column.id);
            columnCards.forEach(card => {
                const cardEl = createCardElement(card);
                cardsContainer.appendChild(cardEl);
            });

            columnDiv.appendChild(header);
            columnDiv.appendChild(cardsContainer);

            return columnDiv;
        }

        /**
         * Create a card element
         */
        function createCardElement(card) {
            const cardDiv = document.createElement('div');
            cardDiv.className = 'kanban-card bg-white p-4 rounded-xl shadow-md border-2 border-slate-100 cursor-move card-enter';
            cardDiv.draggable = true;
            cardDiv.dataset.cardId = card.id;

            // Card content
            const content = document.createElement('p');
            content.className = 'text-slate-700 mb-3 whitespace-pre-wrap';
            content.textContent = card.content;

            // Card actions
            const actions = document.createElement('div');
            actions.className = 'flex gap-2 justify-end';

            // Edit button
            const editBtn = document.createElement('button');
            editBtn.className = 'text-blue-500 hover:text-blue-700 p-2 rounded-lg hover:bg-blue-50 transition-all';
            editBtn.innerHTML = '‚úèÔ∏è';
            editBtn.onclick = (e) => {
                e.stopPropagation();
                openEditCardModal(card);
            };

            // Delete button
            const deleteBtn = document.createElement('button');
            deleteBtn.className = 'text-red-500 hover:text-red-700 p-2 rounded-lg hover:bg-red-50 transition-all';
            deleteBtn.innerHTML = 'üóëÔ∏è';
            deleteBtn.onclick = (e) => {
                e.stopPropagation();
                openDeleteModal(card);
            };

            actions.appendChild(editBtn);
            actions.appendChild(deleteBtn);

            cardDiv.appendChild(content);
            cardDiv.appendChild(actions);

            // Drag event listeners
            cardDiv.addEventListener('dragstart', handleDragStart);
            cardDiv.addEventListener('dragend', handleDragEnd);

            return cardDiv;
        }

        /**
         * Handle drag start event
         */
        function handleDragStart(e) {
            draggedCard = e.target;
            e.target.classList.add('card-dragging');
            e.dataTransfer.effectAllowed = 'move';
            e.dataTransfer.setData('text/html', e.target.innerHTML);
        }

        /**
         * Handle drag end event
         */
        function handleDragEnd(e) {
            e.target.classList.remove('card-dragging');
            
            // Remove drag-over styling from all columns
            document.querySelectorAll('[data-column-id]').forEach(col => {
                col.querySelector('div[data-column-id]')?.classList.remove('column-drag-over');
            });
        }

        /**
         * Handle drag over event
         */
        function handleDragOver(e) {
            if (e.preventDefault) {
                e.preventDefault();
            }
            e.dataTransfer.dropEffect = 'move';
            e.currentTarget.classList.add('column-drag-over');
            return false;
        }

        /**
         * Handle drag leave event
         */
        function handleDragLeave(e) {
            e.currentTarget.classList.remove('column-drag-over');
        }

        /**
         * Handle drop event
         */
        function handleDrop(e) {
            if (e.stopPropagation) {
                e.stopPropagation();
            }
            e.preventDefault();

            e.currentTarget.classList.remove('column-drag-over');

            const targetColumnId = e.currentTarget.dataset.columnId;
            const cardId = draggedCard.dataset.cardId;
            const card = cards.find(c => c.id === cardId);

            if (card && card.columnId !== targetColumnId) {
                const oldColumnId = card.columnId;
                card.columnId = targetColumnId;
                saveCards();
                renderBoard();

                // Trigger confetti if moved to "Done"
                if (targetColumnId === 'done') {
                    triggerConfetti();
                }
            }

            return false;
        }

        /**
         * Open modal to add a new card
         */
        function openAddCardModal(columnId) {
            currentModalColumn = columnId;
            currentEditingCard = null;
            document.getElementById('modal-title').textContent = 'Add New Card';
            document.getElementById('card-input').value = '';
            document.getElementById('card-modal').classList.remove('hidden');
            document.getElementById('card-input').focus();
        }

        /**
         * Open modal to edit an existing card
         */
        function openEditCardModal(card) {
            currentEditingCard = card;
            currentModalColumn = card.columnId;
            document.getElementById('modal-title').textContent = 'Edit Card';
            document.getElementById('card-input').value = card.content;
            document.getElementById('card-modal').classList.remove('hidden');
            document.getElementById('card-input').focus();
        }

        /**
         * Close the card modal
         */
        function closeCardModal() {
            document.getElementById('card-modal').classList.add('hidden');
            currentEditingCard = null;
            currentModalColumn = null;
        }

        /**
         * Save card (add or edit)
         */
        function saveCard() {
            const content = document.getElementById('card-input').value.trim();
            
            if (!content) {
                return;
            }

            if (currentEditingCard) {
                // Edit existing card
                currentEditingCard.content = content;
            } else {
                // Add new card
                const newCard = {
                    id: generateId(),
                    content: content,
                    columnId: currentModalColumn
                };
                cards.push(newCard);
            }

            saveCards();
            renderBoard();
            closeCardModal();
        }

        /**
         * Open delete confirmation modal
         */
        function openDeleteModal(card) {
            currentDeletingCard = card;
            document.getElementById('delete-modal').classList.remove('hidden');
        }

        /**
         * Close delete confirmation modal
         */
        function closeDeleteModal() {
            document.getElementById('delete-modal').classList.add('hidden');
            currentDeletingCard = null;
        }

        /**
         * Delete the card
         */
        function deleteCard() {
            if (currentDeletingCard) {
                cards = cards.filter(c => c.id !== currentDeletingCard.id);
                saveCards();
                renderBoard();
                closeDeleteModal();
            }
        }

        /**
         * Setup event listeners for modals
         */
        function setupEventListeners() {
            // Card modal
            document.getElementById('save-card-btn').addEventListener('click', saveCard);
            document.getElementById('cancel-card-btn').addEventListener('click', closeCardModal);
            
            // Delete modal
            document.getElementById('confirm-delete-btn').addEventListener('click', deleteCard);
            document.getElementById('cancel-delete-btn').addEventListener('click', closeDeleteModal);

            // Close modals on escape key
            document.addEventListener('keydown', (e) => {
                if (e.key === 'Escape') {
                    closeCardModal();
                    closeDeleteModal();
                }
            });

            // Close modals on backdrop click
            document.getElementById('card-modal').addEventListener('click', (e) => {
                if (e.target.id === 'card-modal') {
                    closeCardModal();
                }
            });

            document.getElementById('delete-modal').addEventListener('click', (e) => {
                if (e.target.id === 'delete-modal') {
                    closeDeleteModal();
                }
            });

            // Save card on Ctrl/Cmd + Enter
            document.getElementById('card-input').addEventListener('keydown', (e) => {
                if ((e.ctrlKey || e.metaKey) && e.key === 'Enter') {
                    saveCard();
                }
            });
        }

        /**
         * Trigger confetti animation
         */
        function triggerConfetti() {
            const colors = ['#ff0000', '#00ff00', '#0000ff', '#ffff00', '#ff00ff', '#00ffff', '#ffa500', '#ff1493'];
            const confettiCount = 50;

            for (let i = 0; i < confettiCount; i++) {
                setTimeout(() => {
                    createConfetti(colors[Math.floor(Math.random() * colors.length)]);
                }, i * 30);
            }
        }

        /**
         * Create a single confetti piece
         */
        function createConfetti(color) {
            const confetti = document.createElement('div');
            confetti.className = 'confetti';
            confetti.style.backgroundColor = color;
            confetti.style.left = Math.random() * window.innerWidth + 'px';
            confetti.style.top = '-10px';
            confetti.style.transform = `rotate(${Math.random() * 360}deg)`;
            
            document.body.appendChild(confetti);

            const duration = 2000 + Math.random() * 1000;
            const endX = (Math.random() - 0.5) * 200;
            const endY = window.innerHeight + 10;

            confetti.animate([
                { transform: `translate(0, 0) rotate(0deg)`, opacity: 1 },
                { transform: `translate(${endX}px, ${endY}px) rotate(${Math.random() * 720}deg)`, opacity: 0 }
            ], {
                duration: duration,
                easing: 'cubic-bezier(0.25, 0.46, 0.45, 0.94)'
            }).onfinish = () => {
                confetti.remove();
            };
        }

        // Initialize the application when DOM is ready
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', init);
        } else {
            init();
        }
    </script>
</body>
